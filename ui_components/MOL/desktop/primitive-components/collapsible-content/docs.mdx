### Description

The Collapsible Component is a dynamic height container that renders a scrollable content in the Exploration UI. The content can be collapsed/expanded by clicking toggle button.

<ComponentProps
	items={[
		{
			propName: 'isContentOpen',
			type: '<boolean>*',
			defaultValue: 'undefined',
			description: (
				<p>
					Component toggle state. If <code>true</code> then component will be
					expanded. If <code>false</code> then component will be collapsed.
				</p>
			),
		},
		{
			propName: 'onCollapseButtonClick',
			type: '<function>*',
			defaultValue: 'undefined',
			description: (
				<p>The function to be called when the collapse button is clicked.</p>
			),
		},
		{
			propName: 'collapsePanelTooltipContent',
			type: '<string>*',
			defaultValue: 'undefined',
			description: (
				<p>
					The content used for the tooltip that appears on the collapse button
					when content is expanded.
				</p>
			),
		},
		{
			propName: 'expandPanelTooltipContent',
			type: '<string>*',
			defaultValue: 'undefined',
			description: (
				<p>
					The content used for the tooltip that appears on the collapse button
					when content is collapsed.
				</p>
			),
		},
		{
			propName: 'hasCustomScroll',
			type: '<boolean>',
			defaultValue: 'false',
			description: (
				<p>
					Set to <code>true</code> if scroll handler should be disabled and
					scroll detection will happen externally. If set to <code>true</code>{' '}
					then scroll overflow on the content will be disabled and if desired
					has to be handled in children component.
				</p>
			),
		},
		{
			propName: 'isContentScrolled',
			type: '<boolean>',
			defaultValue: 'false',
			description: (
				<p>
					Set to <code>true</code> if scroll detection happened externally and
					inner shadow should be applied.
				</p>
			),
		},
		{
			propName: 'hasHeightOverride',
			type: '<boolean>',
			defaultValue: 'false',
			description: (
				<p>
					By default Collapsible Content height will be automatically adjusted
					based of the inner content height. If you set this prop to
					<code>true</code> then content will stretch to fill 100% of the available
					space and will not adjust it's height based on the content height. If you
					want to change the Collapsible Content height pass it through <code>
						height
					</code> prop.
				</p>
			),
		},
	]}
/>
